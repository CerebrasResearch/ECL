\subsection{Proof of \texorpdfstring{\cref{thm:sobol_equiv}}{Theorem: Sobol Equivalence}}

\begin{proof}
\textbf{Part (a): vector identity.} Under the additive vector model $f(X) = \sum_{i=1}^L g_i(X_i)$ with independent coordinates and independent-copy perturbation at position $i$:
\[
f(X) - f(X^{(i)}) = g_i(X_i) - g_i(X_i'),
\]
where $X_i' \sim P_i$ is independent of $X_i$. Therefore:
\[
\cI(i;r) = \E\!\left[\norm{g_i(X_i) - g_i(X_i')}_2^2\right] = 2\,\mathrm{tr}(\Cov(g_i(X_i))),
\]
using the identity $\E[\norm{Y-Y'}_2^2] = 2\,\mathrm{tr}(\Cov(Y))$ for i.i.d.\ vector copies $Y, Y'$. Defining $V_{\mathrm{tot}}^{\mathrm{vec}} := \mathrm{tr}(\Cov(f(X))) = \sum_i \mathrm{tr}(\Cov(g_i(X_i)))$ (using independence), the vector total-effect index $S_i^{T,\mathrm{vec}} := \mathrm{tr}(\Cov(g_i(X_i)))/V_{\mathrm{tot}}^{\mathrm{vec}}$ satisfies $\cI(i;r) = 2\,V_{\mathrm{tot}}^{\mathrm{vec}}\, S_i^{T,\mathrm{vec}}$.

\textbf{Part (b): scalar projection.} Fix any scalar projection $\varphi:\cZ \to \bbR$ and define $Y := \varphi(f(X))$, $V_{\mathrm{tot}} := \Var(Y)$, and the projected single-site influence
\[
\cI^\varphi(i;r) := \E[(Y - \varphi(f(X^{(i)})))^2].
\]
By the same calculation applied to the scalar variable $\varphi(g_i(X_i))$:
\[
\cI^\varphi(i;r) = \E\!\left[(\varphi(g_i(X_i)) - \varphi(g_i(X_i')))^2\right] = 2\,\Var(\varphi(g_i(X_i))).
\]
For an additive model with independent coordinates, the total variance of $Y$ decomposes as $V_{\mathrm{tot}} = \sum_i \Var(\varphi(g_i(X_i)))$, and the Sobol total-effect index for coordinate $i$ equals the first-order index $S_i^T = \Var(\varphi(g_i(X_i)))/V_{\mathrm{tot}}$. Therefore $\cI^\varphi(i;r) = 2\,V_{\mathrm{tot}}\, S_i^T$.
\end{proof}

\subsection{Quantitative tail approximation (\texorpdfstring{\cref{prop:tail_add}}{Proposition: Tail Approximation})}

\begin{proof}
Let $f(X)=\sum_{i=1}^L g_i(X_i) + \epsilon(X)$ and $S = \{i:|i-r|>\ell\}$. Define:
\begin{align*}
A_S &:= \sum_{i\in S}(g_i(X_i)-g_i(X_i')), \\
B_S &:= \epsilon(X)-\epsilon(X^{(S)}).
\end{align*}
Then $\Delta_S := f(X) - f(X^{(S)}) = A_S + B_S$ and:
\[
\E\norm{\Delta_S}^2 = \E\norm{A_S}^2 + 2\E\ip{A_S}{B_S} + \E\norm{B_S}^2.
\]
Under independence and the additive structure:
\[
\E\norm{A_S}^2 = \sum_{i\in S}\E\norm{g_i(X_i)-g_i(X_i')}^2 = \sum_{i\in S} \cI(i;r).
\]
By Cauchy--Schwarz: $|2\E\ip{A_S}{B_S}| \le 2\sqrt{\E\norm{A_S}^2}\sqrt{\E\norm{B_S}^2}$.

To bound $\E\norm{B_S}^2$ in terms of $\eta^2$, write $B_S = \epsilon(X) - \epsilon(X^{(S)})$ and apply the parallelogram bound: $\norm{a-b}^2 \le 2(\norm{a}^2 + \norm{b}^2)$, giving $\E\norm{B_S}^2 \le 2(\E\norm{\epsilon(X)}^2 + \E\norm{\epsilon(X^{(S)})}^2)$. Since $X^{(S)} \stackrel{d}{=} X$ under the perturbation kernel preserving the marginal of $\epsilon$ (or otherwise bounded by the same hypothesis), $\E\norm{\epsilon(X^{(S)})}^2 \le \eta^2$, yielding $\E\norm{B_S}^2 \le 4\eta^2$.

Combining,
\[
\left|\Delta_{\mathrm{out}}(\ell;r) - \sum_{i\in S}\cI(i;r)\right| \le 2\sqrt{\sum_{i\in S}\cI(i;r)} \cdot 2\eta + 4\eta^2.
\]
\end{proof}

\subsection{Variance decomposition identity}

For completeness, we record the law of total variance:
\[
\Var(Y) = \E[\Var(Y\mid \cG)] + \Var(\E[Y\mid \cG]),
\]
where $\cG$ is a sub-$\sigma$-algebra. This underlies the Sobol decomposition: conditioning on $X_S$ yields $V_S := \Var(\E[Y\mid X_S])$, the variance explained by coordinates in $S$.

\subsection{Explicit computation for quadratic models}

Consider $f(X) = \sum_i g_i(X_i) + \sum_{i<j} g_{ij}(X_i, X_j)$ (pairwise interactions). The influence energy for perturbing position $i$ becomes:
\[
\cI(i;r) = 2\Var(g_i(X_i)) + 2\sum_{j \neq i} \E\!\left[\Var(g_{ij}(X_i, X_j) \mid X_j)\right],
\]
where the second term captures the interaction contribution. This shows that single-site perturbation captures both main effects and the ``average interaction'' with all other positions, but not the specific pairwise interaction structure.

\subsection{Whitening invariance}

Let $\Sigma=\Cov(f(X))$ be positive definite and define whitened embeddings $\tilde f(X)=\Sigma^{-1/2}(f(X)-\E[f(X)])$. For any invertible linear transform $T$, the model $Tf$ has covariance $T\Sigma T^\top$. Whitening yields $(T\Sigma T^\top)^{-1/2}(Tf(X) - T\E[f(X)]) = (T\Sigma T^\top)^{-1/2}T\Sigma^{1/2}\tilde{f}(X) = Q\tilde f(X)$, where $Q := (T\Sigma T^\top)^{-1/2}T\Sigma^{1/2}$. To verify orthogonality, compute
\[
QQ^\top \;=\; (T\Sigma T^\top)^{-1/2}\,T\Sigma^{1/2}\,\Sigma^{1/2}T^\top\,(T\Sigma T^\top)^{-1/2}
\;=\; (T\Sigma T^\top)^{-1/2}\,T\Sigma T^\top\,(T\Sigma T^\top)^{-1/2}
\;=\; I.
\]
Hence $Q$ is orthogonal, and ECL under Mahalanobis distance is invariant under invertible linear reparameterization.
